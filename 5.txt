{

void printDigitsReversed(int num)
{
	int remainder=0;
	while (num > 0)
	{
		remainder = num % 10;
		num = num / 10;
		cout << remainder;
	}


}

{
	printDigitsReversed(ReadPositiveNumber("Please Enter Positive Number\n"));
	return 0;
}
