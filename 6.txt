void print_Sum_Of_Digits(int num)
{
	int remainder = 0; int sum=0;
	while (num > 0)
	{
		remainder = num % 10;
		num = num / 10;
		sum += remainder;
	}
	cout << "Sum Of Digits is: " << sum;


}
