#include <iostream>        
#include <string>           
using namespace std;            

int ReadPositiveNumber(string Message)
{
    int Number = 0;        
    do
    {
        cout << Message << endl;      
        cin >> Number;                   
    } while (Number <= 0);               

    return Number;       
}


void InvertedNumberPattern(int num)
{
    for (int i = num; i > 0; i--)
    {
        for (int j = i; j > 0; j--)
            cout << i;
        cout << endl;
    }
}
void NormalNumberPattern(int num)
{
    for (int i = 1; i <= num; i++)
    {
        for (int j = i; j > 0; j--)
        {
            cout << i;
        }
        cout << endl;
    }




}
void LatterPattern(int Number)
{
    cout << "\n";        

    for (int i = 65; i <= 65 + Number - 1; i++)
    {
        for (int j = 1; j <= Number-(65+Number-1-i); j++)
        {
            cout << char(i);
        }
        cout << endl;
   }
}


int main()
{
    LatterPattern(ReadPositiveNumber("Please Enter Number : "));
    return 0;
}
